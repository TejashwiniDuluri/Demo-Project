5 . Defining models in models.py

from django.db import models

Class Employee (models.Model ) :

name = models.CharField(max_length = 20)

age = models. IntegerField ( default=0 )

address = models.TextField ( )

def __ str __ ( self ) :

return self .name

Field Types :

String or Character : model.CharField ( max_length = 20)

Integer : models.IntegerField( default = 0)

True or False : models.BooleanField ( )

Text : models.TextField ( )

Floating Numbers : models.FloatField ( default = 0.0)

Decimal numbers : models.DecimalField(max_digits = 20 , decimal places = 4)

ForeignKey : models.ForeignKey ( ‘model_name’ , on_delete = models.CASCADE )
